FUNCTION_BLOCK "Burner"  // SCL  Block #13

VAR_INPUT

    FlameBurn : Bool;  // Пламя горит
END_VAR
VAR_OUTPUT
    TL : Bool;  // Включение горелки
    TR : Bool;  // Управление мощностью
    RS : Bool;  // Дистанционный перезапуск
END_VAR
VAR
    DataHMI : "BurnerHMI";  // Ручной режим
    interlockCommon : Bool;  // Блокировка
    enableCommon : Bool;
    AutoStart : Bool;  // Авто. режим. Запуск горелки
    AutoUpPower : Bool;  // Авто. режим. Увеличить мощность
    AutoAlarmReset : Bool;  // Авто. режим. Сброс аварии
    UpPower : Bool;  // Увеличить мощность
    CounterAlarm : Int;  // Счетчик аварий
    ton_Reset : TON_TIME;  // Время ожидания сброса аварии
    FatalError : Bool;  // Ошибка после попыток авто. перезапуска
    rt_Alarm : R_TRIG;
    setTonReset : Bool;
END_VAR
VAR CONSTANT
    maxTry : Int := 3;  // Максимальное кол. попыток перезапуска
    timeWaitReset : Time := T#30s;
END_VAR

// ==================== CODE ====================
// Network 1
// Горелка

REGION Аварийный перезапуск
    // Только в автоматическом режиме

    IF DataHMI.ControlMode =  THEN
        // Если еще не пробовали производить сброс

        IF NOT FatalError THEN
            // Счетчик попыток перезапуска при ошибке

            rt_Alarm(CLK := Alarm);
            IF rt_Alarm.Q THEN
                CounterAlarm := CounterAlarm + ;
            END_IF;
            IF CounterAlarm <>  AND Alarm THEN
                setTonReset := ;
            ELSE
                setTonReset := ;
            END_IF;
            // Запуск после задержки

            ton_Reset(IN := setTonReset,
                       PT := ,
                       Q => AutoAlarmReset);
            // Сброс попыток

            IF CounterAlarm >=  THEN
                CounterAlarm := ;
                FatalError := ;
            END_IF;
        END_IF;
    ELSE
        RESET_TIMER(ton_Reset);
        FatalError := ;
        setTonReset := ;
    END_IF;
END_REGION
REGION Сброс ручного управления
    IF NOT Alarm AND DataHMI.ControlMode <>  THEN
        DataHMI.ManualStart := ;
        DataHMI.ManualUpPower := ;
        DataHMI.ManualAlarmReset := ;
    END_IF;
END_REGION
REGION Блокировка
    interlockCommon := enableCommon OR NOT Alarm;
END_REGION
REGION Управление горелкой
    TL := NOT interlockCommon AND NOT Alarm AND MUX(K := DataHMI.ControlMode,
                                          IN0 := ,
                                          IN1 := DataHMI.ManualStart,
                                          IN2 :=  AutoStart);
    TR := NOT interlockCommon AND FlameBurn AND MUX(K := DataHMI.ControlMode,
                                 IN0 := ,
                                 IN1 := DataHMI.ManualUpPower,
                                 IN2 := AutoUpPower);
    RS := MUX(K := DataHMI.ControlMode,
                                IN0 := ,
                                IN1 := DataHMI.ManualAlarmReset,
                                IN2 := AutoAlarmReset);
END_REGION

FUNCTION_BLOCK "Motor"  // SCL  Block #5

VAR_INPUT
    FeedbackForward : Bool;  // Обратная связь. Прямой пуск
    FeedbackOverload : Bool;  // Обратная связь. Перегруз
    FeedbackOverheat : Bool;  // Обратная связь. Перегрев
END_VAR
VAR_OUTPUT
    CommandForward : Bool;  // Управление. Прямой пуск
END_VAR
VAR
    DataHMI : "MotorHMI";  // Данные HMI
    EnableCommon : Bool;  // Условие разрешения работы исполнительного механизма
    InterlockCommon : Bool;  // Блокировка  исполнительного механизма
    AutoForward : Bool;  // Автоматический режим. Прямой пуск
    AlarmCommon : Bool;  // Авария. Общая неисправность
    AlarmForward : Bool;  // Авария. Превышено время пуска
    AlarmOverload : Bool;  // Авария. Перегрузка
    AlarmOverheat : Bool;  // Авария. Перегерев
    AlarmTimer : TON_TIME;  // Таймер формирования аварии
    AlarmTimeSet : Time := T#15S;  // Время запуска. Формирование аварии. Фактическое значение
    RunningTimeCounter : "RUNNING_TIMER";  // Счетчик времени наработки
END_VAR

// ==================== CODE ====================
// Network 1
// Двигатель. Прямой пуск. Входные сигналы: Прямой пуск, Перегрузка, Перегрев

REGION Квитирование
    IF DataHMI.Acknowledge THEN
        AlarmForward := AlarmOverload := AlarmOverheat := ;
    END_IF;
END_REGION
REGION Аварии
    // Авария. Превышено время пуска

    AlarmTimer(IN := CommandForward AND NOT FeedbackForward,
                PT := AlarmTimeSet);
    IF AlarmTimer.Q THEN
        AlarmForward := ;
    END_IF;
    // Авария. Перегрузка

    IF NOT FeedbackOverload THEN
        AlarmOverload := ;
    END_IF;
    // Авария. Перегрев

    IF NOT FeedbackOverheat THEN
        AlarmOverheat := ;
    END_IF;
    // Авария. Общая неисправность

    AlarmCommon := AlarmForward OR AlarmOverload OR AlarmOverheat;
END_REGION
REGION Блокировка
    InterlockCommon := NOT EnableCommon OR AlarmCommon;
END_REGION
REGION Сброс ручного управления
    IF InterlockCommon OR DataHMI.ControlMode <>  THEN
        DataHMI.ManualForward := ;
    END_IF;
END_REGION
REGION Управление
    CommandForward := NOT InterlockCommon AND MUX(K := DataHMI.ControlMode,
                                                    IN0 := ,
                                                    IN1 := DataHMI.ManualForward,
                                                    IN2 := AutoForward);
END_REGION
REGION Слово состояния
    DataHMI.StatusWord := CommandForward;
    DataHMI.StatusWord := FeedbackForward;
    DataHMI.StatusWord := AlarmCommon;
    DataHMI.StatusWord := AlarmForward;
    DataHMI.StatusWord := AlarmOverload;
    DataHMI.StatusWord := AlarmOverheat;
    DataHMI.StatusWord := InterlockCommon;
    DataHMI.StatusWord := DataHMI.ControlMode = ;
END_REGION
REGION Индикатор
    DataHMI.StatusIndicator := AlarmCommon;
    DataHMI.StatusIndicator := FeedbackForward;
END_REGION
REGION Время наработки
    (Enable:=FeedbackForward,
                        ResetTime := DataHMI.RunningTimeReset,
                        RunningTime := DataHMI.RunningTime);
END_REGION

FUNCTION_BLOCK "MotorVFD"  // SCL  Block #7

VAR_INPUT

VAR
    DataHMI : "MotorVFDHMI";  // Данные HMI
    VFDData : "Inovance";  // Данные ПЧ
    EnableCommon : Bool;  // Условие разрешения работы исполнительного механизма
    EnableForward : Bool;  // Условие разрешения прямого пуска
    EnableReverse : Bool;  // Условие разрешения реверсивного пуска
    InterlockCommon : Bool;  // Блокировка  исполнительного механизма
    SetpointSpeed : Real;  // Задание скорости
    AutoSpeed : Real;  // Автоматический режим. Задание скорости
    AutoForward : Bool;  // Автоматический режим. Прямой пуск
    AutoReverse : Bool;  // Автоматический режим. Реверсивный пуск
    FeedbackFault : Bool;  // Обратная связь. Неисправность
    FeedbackForward : Bool;  // Обратная связь. Прямой пуск
    FeedbackReverse : Bool;  // Обратная связь. Реверсивный пуск
    FeedbackLink : Bool;  // Обратная связь. Обмен данными
    FeedbackLocal : Bool;  // Обратная связь. Локальное управление
    AlarmCommon : Bool;  // Авария. Общая неисправность
    AlarmFault : Bool;  // Авария. Неисправность ПЧ
    AlarmForward : Bool;  // Авария. Превышено время прямого пуска
    AlarmReverse : Bool;  // Авария. Превышено время реверсивного пуска
    AlarmOverload : Bool;  // Авария. Перегрузка
    AlarmOverheat : Bool;  // Авария. Перегерев
    AlarmLink : Bool;  // Авария. Обмен данными
    AlarmSpeed : Bool;  // Авария датчика скорости
    AlarmTimer : TON_TIME;  // Таймер 1 секунда
    AlarmTimeSet : Int := 15;  // Время запуска. Формирование аварии. Фактическое значение
    RunningTimeCounter : "RUNNING_TIMER";  // Счетчик времени наработки
    SpeedMinFrq : Real;  // Минимальная частота ПЧ, Гц
    SpeedMaxFrq : Real := 50.0;  // Максимальная частота ПЧ, Гц
    StatusCmdFwd : Bool;  // Статус команды прямого пуска
    StatusCmdRev : Bool;  // Статус команды реверсивного пуска

VAR_TEMP
    ReadDataCode : Int;
    WriteDataCode : Int;
    SpeedSet : Real;  // Задание частоты, %

VAR CONSTANT
    SpeedSetMin : Real := 40.0;  // Минимальное задаваемое значение частоты в процентах
    SpeedSetMax : Real := 100.0;  // Максимальное задаваемое значение частоты в процентах
    SpeedLimitMin : Real := 0.0;  // Минимальное значение частоты в процентах
    SpeedLimitMax : Real := 100.0;  // Максимальное значение частоты в процентах
    ControlWordRunFwd : Word := 16#0001;  // Слово управления - Прямой пуск
    ControlWordRunRev : Word := 16#0002;  // Слово управления - Реверсивный пуск
    ControlWordJogFwd : Word := 16#0003;  // Слово управления - Пуск
    ControlWordJogRev : Word := 16#0004;  // Слово управления - Стоп
    ControlWordStopCoast : Word := 16#0005;  // Слово управления - Останов выбегом
    ControlWordStopMode : Word := 16#0006;  // Слово управления - Останов выбранным режимом
    ControlWordAck : Word := 16#0007;  // Слово управления - Квитирование
END_VAR

// ==================== CODE ====================
// Network 1
// Двигатель. ПЧ. Входные сигналы: Перегрузка, Перегрев, Датчик скорости, Местный/Дист режим

REGION Чтение слова состояния и текущей скорости
    ReadDataCode := DPRD_DAT(LADDR := DeviceAddress, RECORD => VFDData.Status);
    // #FeedbackLocal := статус локального управления;

    IF NOT FeedbackLocal THEN
        FeedbackFault := VFDData.Status.FaultStatus;
        FeedbackForward := VFDData.Status.RunningStatus AND NOT VFDData.Status.RunningDirection;
        FeedbackReverse := VFDData.Status.RunningStatus AND VFDData.Status.RunningDirection;
        FeedbackLink := ReadDataCode = ;
        // #DataHMI.ActualSpeed := #VFDData.Status.SpeedActual * 2 / 100;

    END_IF;
END_REGION
REGION Запись слова управление и уставки скорости
    WriteDataCode := DPWR_DAT(LADDR := DeviceAddress, RECORD := VFDData.Control);
END_REGION
REGION Слово управления
    IF CommandForward THEN
        VFDData.Control.ControlWord := ;
    ELSIF CommandReverse THEN
        VFDData.Control.ControlWord := ;
    ELSIF DataHMI.Acknowledge THEN
        VFDData.Control.ControlWord := ;
    ELSE
        VFDData.Control.ControlWord := ;
    END_IF;
// IF #Local_Remote THEN
        #VFDData.Control.ControlWord := управление с клемм;
    END_IF;

END_REGION
// Проверка локальное / дистанционное управление

IF Local_Remote THEN
    // Дистанционное управление

    REGION Квитирование
        IF DataHMI.Acknowledge THEN
            AlarmFault := AlarmForward := AlarmReverse := AlarmOverload := AlarmOverheat := AlarmLink := AlarmSpeed := ;
        END_IF;
    END_REGION
    REGION Аварии
        // Обратная связь. Неисправность

        IF FeedbackFault THEN
            AlarmFault := ;
        END_IF;
        // Авария. Превышено время пуска

        AlarmTimer(IN := (CommandForward AND NOT FeedbackForward)
                    OR (CommandReverse AND NOT FeedbackReverse),
                    PT := INT_TO_TIME(AlarmTimeSet) * );
        // Авария. Превышено время прямого пуска

        IF AlarmTimer.Q AND CommandForward AND NOT FeedbackForward THEN
            AlarmForward := ;
        END_IF;
        // Авария. Превышено время реверсивного пуска

        IF AlarmTimer.Q AND CommandReverse AND NOT FeedbackReverse THEN
            AlarmReverse := ;
        END_IF;
        // Авария. Перегрузка

        IF NOT FeedbackOverload THEN
            AlarmOverload := ;
        END_IF;
        // Авария. Перегрев

        IF NOT FeedbackOverheat THEN
            AlarmOverheat := ;
        END_IF;
        // Авария. Обмен данными

        IF NOT FeedbackLink THEN
            AlarmLink := ;
        END_IF;
        // Авария. Датчик скорости не считает импульсы

        IF NOT FeedbackSpeed AND (FeedbackForward OR FeedbackReverse) THEN
            AlarmSpeed := ;
        END_IF;
        
        // Авария. Общая неисправность

        AlarmCommon := AlarmFault OR AlarmForward OR AlarmReverse OR AlarmOverload OR AlarmOverheat OR AlarmLink OR AlarmSpeed;
    END_REGION
    REGION Блокировка
        InterlockCommon := NOT EnableCommon OR AlarmCommon;
    END_REGION
    REGION Сброс ручного управления
        IF InterlockCommon OR DataHMI.ControlMode <>  OR NOT EnableForward THEN
            DataHMI.ManualForward := ;
        END_IF;
        IF InterlockCommon OR DataHMI.ControlMode <>  OR NOT EnableReverse THEN
            DataHMI.ManualReverse := ;
        END_IF;
    END_REGION
    REGION Управление
        StatusCmdFwd := NOT InterlockCommon AND EnableForward AND MUX(K := DataHMI.ControlMode,
                                                                         IN0 := ,
                                                                         IN1 := DataHMI.ManualForward,
                                                                         IN2 := AutoForward);
        StatusCmdRev := NOT InterlockCommon AND EnableReverse AND MUX(K := DataHMI.ControlMode,
                                                                         IN0 := ,
                                                                         IN1 := DataHMI.ManualReverse,
                                                                         IN2 := AutoReverse);
        CommandForward := StatusCmdFwd;
        CommandReverse := StatusCmdRev;
    END_REGION
    REGION Уставка скорости         
        IF DataHMI.ControlMode =  THEN
            SpeedSet := DataHMI.ManualSpeed := LIMIT(MN := , IN := DataHMI.ManualSpeed, MX := );
        ELSIF DataHMI.ControlMode =  THEN
            SpeedSet := AutoSpeed := LIMIT(MN := , IN := AutoSpeed, MX := );
        END_IF;
        VFDData.Control.SpeedSetpoint := REAL_TO_INT(SCALE_X(MIN := SpeedMinFrq, VALUE := NORM_X(MIN := , VALUE := SpeedSet, MAX := ), MAX := SpeedMaxFrq) * );
    END_REGION
    REGION Актуальная скорость по датчику скорости
        IF NOT AlarmSpeed THEN
            DataHMI.ActualSpeed := Speed;
        ELSE
            DataHMI.ActualSpeed := ;
        END_IF;
    END_REGION
    REGION Слово состояния
        DataHMI.StatusWord := CommandForward;
        DataHMI.StatusWord := CommandReverse;
        DataHMI.StatusWord := FeedbackForward;
        DataHMI.StatusWord := FeedbackReverse;
        DataHMI.StatusWord := AlarmCommon;
        DataHMI.StatusWord := AlarmFault;
        DataHMI.StatusWord := AlarmForward;
        DataHMI.StatusWord := AlarmReverse;
        DataHMI.StatusWord := AlarmOverload;
        DataHMI.StatusWord := AlarmOverheat;
        DataHMI.StatusWord := AlarmLink;
        DataHMI.StatusWord := InterlockCommon;
        DataHMI.StatusWord := AlarmSpeed;
        DataHMI.StatusWord := DataHMI.ControlMode = ;
    END_REGION
    REGION Индикатор
        DataHMI.StatusIndicator := AlarmCommon;
        DataHMI.StatusIndicator := CommandForward OR CommandReverse;
    END_REGION
    REGION Направление
        DataHMI.StatusDirection := FeedbackForward;
        DataHMI.StatusDirection := FeedbackReverse;
    END_REGION
    REGION Время наработки
        (Enable := FeedbackForward OR FeedbackReverse,
                            ResetTime := DataHMI.RunningTimeReset,
                            RunningTime := DataHMI.RunningTime);
    END_REGION
ELSE
    // Статус локального управления

    DataHMI.StatusWord := Local_Remote;
END_IF;

    AIMax : Int := 27648;  // Максимальное значение входного сигнала

    SensorCurrentMax : Real := 20.0;  // Верхнее токовое значние датчика
    ChortCircuitLimit : Real := 0.0125;  // Коэффициент для тока короткого замыкания
    WireBreakLimit : Real := 0.0125;  // Коэффициент для тока обрыва датчика

// ==================== CODE ====================
// Network 1
// Обработка аналогового сигнала

REGION Check Data // Проверка корректности вводимых данных
    IF (DataHMI.ScaleMax < DataHMI.ScaleMin)
    THEN
        Templimit := DataHMI.ScaleMax;
        DataHMI.ScaleMax := DataHMI.ScaleMin;
        DataHMI.ScaleMin := Templimit;
    END_IF;
    IF (DataHMI.LLAlarmLimit < DataHMI.ScaleMin) OR
        (DataHMI.LLAlarmLimit > DataHMI.ScaleMax)
    THEN
        DataHMI.LLAlarmLimit := DataHMI.ScaleMin;
    END_IF;
    IF (DataHMI.HHAlarmLimit > DataHMI.ScaleMax) OR
        (DataHMI.HHAlarmLimit <= DataHMI.ScaleMin)
    THEN
        DataHMI.HHAlarmLimit := DataHMI.ScaleMax;
    END_IF;
    IF (DataHMI.LAlarmLimit < DataHMI.ScaleMin) OR
        (DataHMI.LAlarmLimit > DataHMI.ScaleMax)
    THEN
        DataHMI.LAlarmLimit := DataHMI.ScaleMin;
    END_IF;
    IF (DataHMI.HAlarmLimit > DataHMI.ScaleMax) OR
        (DataHMI.HAlarmLimit < DataHMI.ScaleMin)
    THEN
        DataHMI.HAlarmLimit := DataHMI.ScaleMax;
    END_IF;
    IF DataHMI.HAlarmLimit > DataHMI.HHAlarmLimit
    THEN
        Templimit := DataHMI.HAlarmLimit;
        DataHMI.HAlarmLimit := DataHMI.HHAlarmLimit;
        DataHMI.HHAlarmLimit := Templimit;
    END_IF;
    IF DataHMI.LAlarmLimit < DataHMI.LLAlarmLimit
    THEN
        Templimit := DataHMI.LAlarmLimit;
        DataHMI.LAlarmLimit := DataHMI.LLAlarmLimit;
        DataHMI.LLAlarmLimit := Templimit;
    END_IF;
    IF DataHMI.LAlarmLimit > DataHMI.HAlarmLimit
    THEN
        Templimit := DataHMI.LAlarmLimit;
        DataHMI.LAlarmLimit := DataHMI.HAlarmLimit;
        DataHMI.HAlarmLimit := Templimit;
    END_IF;
END_REGION
REGION Scale Signal // Шкалирование значений датчика
    IF NOT SimulateMode
    THEN
        RAnalogInput := INT_TO_REAL(InputSignal - ) / ( - ); // #RAnalogInput := #InputSignal / #AIMax;

    ELSE
        RAnalogInput := INT_TO_REAL(SimulateSignal - ) / ( - ); // #RAnalogInput := #SimulateInput / #AIMax;

    END_IF;
    DataHMI.CurrentValue := RAnalogInput * ( - ) + ;
    DataHMI.ActualValue := RAnalogInput * (DataHMI.ScaleMax - DataHMI.ScaleMin) + DataHMI.ScaleMin;
END_REGION
REGION Calculate Hysteresis
    HMinusHysteresis := DataHMI.HAlarmLimit - DataHMI.Hysteresis;
    HPlusHysteresis := DataHMI.HAlarmLimit + DataHMI.Hysteresis;
    LMinusHysteresis := DataHMI.LAlarmLimit - DataHMI.Hysteresis;
    LPlusHysteresis := DataHMI.LAlarmLimit + DataHMI.Hysteresis;
    HHMinusHysteresis := DataHMI.HHAlarmLimit - DataHMI.Hysteresis;
    HHPlusHysteresis := DataHMI.HHAlarmLimit + DataHMI.Hysteresis;
    LLMinusHysteresis := DataHMI.LLAlarmLimit - DataHMI.Hysteresis;
    LLPlusHysteresis := DataHMI.LLAlarmLimit + DataHMI.Hysteresis;
    WBLimit := DataHMI.ScaleMin - (DataHMI.ScaleMax - DataHMI.ScaleMin) * ;
    SCLimit := DataHMI.ScaleMax + (DataHMI.ScaleMax - DataHMI.ScaleMin) * ;
END_REGION
REGION Calculate Alarms
    REGION Set Alarms Bits
        IF DataHMI.ActualValue > SCLimit THEN
            DataHMI.ShortCircuitAlarm := ;
            DataHMI.WireBreakAlarm := ;
            DataHMI.HHAlarm := ;
            DataHMI.HAlarm := ;
            DataHMI.LAlarm := ;
            DataHMI.LLAlarm := ;
        END_IF;
        IF DataHMI.ActualValue < WBLimit THEN
            DataHMI.WireBreakAlarm := ;
            DataHMI.ShortCircuitAlarm := ;
            DataHMI.HHAlarm := ;
            DataHMI.HAlarm := ;
            DataHMI.LAlarm := ;
            DataHMI.LLAlarm := ;
        END_IF;
        // set HH

        HHRestingTimer1(IN := DataHMI.ActualValue > HHPlusHysteresis AND NOT DataHMI.WireBreakAlarm AND NOT DataHMI.ShortCircuitAlarm,
                      PT := RestingTime,
                      Q => DataHMI.HHAlarm);
        
        // set H

        HRestingTimer2(IN := DataHMI.ActualValue > HPlusHysteresis AND NOT DataHMI.WireBreakAlarm AND NOT DataHMI.ShortCircuitAlarm AND DataHMI.ActualValue < HHPlusHysteresis,
                     PT := RestingTime,
                     Q => DataHMI.HAlarm);
        
        // set L

        LRestingTimer3(IN := DataHMI.ActualValue < LMinusHysteresis AND NOT DataHMI.WireBreakAlarm AND NOT DataHMI.ShortCircuitAlarm AND DataHMI.ActualValue > LLMinusHysteresis,
                     PT := RestingTime,
                     Q => DataHMI.LAlarm);
        
        // set LL

        LLRestingTimer4(IN := DataHMI.ActualValue < LLMinusHysteresis AND NOT DataHMI.WireBreakAlarm AND NOT DataHMI.ShortCircuitAlarm,
                      PT := RestingTime,
                      Q => DataHMI.LLAlarm);
        
    END_REGION
    REGION Reset Alarms Bits
        IF DataHMI.ActualValue < SCLimit
        THEN
            DataHMI.ShortCircuitAlarm := ;
        END_IF;
        IF DataHMI.ActualValue > WBLimit
        THEN
            DataHMI.WireBreakAlarm := ;
        END_IF;
    END_REGION
END_REGION
REGION Set Status Message
    DataHMI.StatusWord := ;
    IF DataHMI.ShortCircuitAlarm THEN
        DataHMI.StatusWord := ; // bit 1

    ELSE
        IF DataHMI.WireBreakAlarm THEN
            DataHMI.StatusWord := ; // bit 6

        ELSE
            IF DataHMI.HHAlarm THEN
                DataHMI.StatusWord := ; // bit 2

            ELSE
                IF DataHMI.HAlarm THEN
                    DataHMI.StatusWord := ; // bit 3

                ELSE
                    IF DataHMI.LAlarm THEN
                        DataHMI.StatusWord := ; // bit 4

                    ELSE
                        IF DataHMI.LLAlarm THEN
                            DataHMI.StatusWord := ; // bit 5

                        ELSE
                            DataHMI.StatusWord := ;
                        END_IF;
                    END_IF;
                END_IF;
            END_IF;
        END_IF;
    END_IF;
END_REGION

REGION Конечный автомат
    CASE Step OF
            
            // -------------------------------------------------------

            // ШАГ 0: ОЖИДАНИЕ

            // -------------------------------------------------------

        :
            DrumReady := ;
            RESET_TIMER(TonStartFan);
            RESET_TIMER(TonStartDrum);
            RESET_TIMER(TonPreheat);
            RESET_TIMER(TonUnload);
            RESET_TIMER(TonCooldown);
            
            // Запуск прогрева по кнопке оператора

            IF DataHMI.StartPreheat THEN
                DataHMI.StartPreheat := ;
                DataHMI.StepAlarmCode := ;
                
                // Барабан ещё тёплый — пропускаем прогрев

                IF WarmAfterStop THEN
                    Step := ;
                    // После пуска вентилятора и барабана сразу READY

                    // (флаг WarmAfterStop проверяется в STEP_START_DRUM)

                ELSE
                    Step := ;
                END_IF;
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 1: ПУСК ВЕНТИЛЯТОРА

            // -------------------------------------------------------

        :
            // Ждём обратную связь от ПЧ вентилятора

            TonStartFan(IN := NOT FanFeedback,
                         PT := );
            
            IF FanFeedback THEN
                // Вентилятор запустился — переходим к пуску барабана

                RESET_TIMER(TonStartFan);
                Step := ;
                
            ELSIF TonStartFan.Q THEN
                // Таймаут — авария

                Step := ;
                DataHMI.StepAlarmCode := ;
                RESET_TIMER(TonStartFan);
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 2: ПУСК БАРАБАНА

            // -------------------------------------------------------

        :
            // Ждём обратную связь от ПЧ барабана

            TonStartDrum(IN := NOT DrumFeedback,
                          PT := );
            
            IF DrumFeedback THEN
                RESET_TIMER(TonStartDrum);
                // Барабан тёплый — сразу в READY, прогрев не нужен

                IF WarmAfterStop THEN
                    Step := ;
                ELSE
                    Step := ;
                END_IF;
                
            ELSIF TonStartDrum.Q THEN
                // Таймаут — авария

                Step := ;
                DataHMI.StepAlarmCode := ;
                RESET_TIMER(TonStartDrum);
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 3: ПРОГРЕВ

            // -------------------------------------------------------

        :
            // Горелка включена, ждём таймер прогрева

            TonPreheat(IN := BurnerFlame,
                        PT := DataHMI.PreheatTime);
            
            IF TonPreheat.Q THEN
                // Прогрев завершён

                RESET_TIMER(TonPreheat);
                Step := ;
            END_IF;
            
            // Авария горелки во время прогрева — в FAULT

            IF BurnerFatalError THEN
                Step := ;
                DataHMI.StepAlarmCode := ;
                RESET_TIMER(TonPreheat);
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 4: ГОТОВ К РАБОТЕ

            // -------------------------------------------------------

        :
            DrumReady := ;
            
            // Верхняя логика подала сигнал — продукт пошёл

            IF SystemRunning THEN
                Step := ;
            END_IF;
            
            // Останов оператором

            IF DataHMI.StopDrum THEN
                DataHMI.StopDrum := ;
                DrumReady := ;
                Step := ;
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 5: РАБОТА

            // -------------------------------------------------------

        :
            
            // Авария горелки (после 3 попыток авто перезапуска)

            IF BurnerFatalError THEN
                DataHMI.StepAlarmCode := ;
                Step := ;
            END_IF;
            
            // Авария температуры на входе (не снизилась за таймер)

            IF TempInAlarm THEN
                DataHMI.StepAlarmCode := ;
                Step := ;
            END_IF;
            
            // Авария вентилятора — горелку выкл, продукт выгружаем

            IF FanAlarm THEN
                DataHMI.StepAlarmCode := ;
                Step := ;
            END_IF;
            
            // Нормальный останов оператором

            IF DataHMI.StopDrum THEN
                DataHMI.StopDrum := ;
                DrumReady := ;
                Step := ;
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 6: АВАРИЙНАЯ ВЫГРУЗКА ПРОДУКТА

            // -------------------------------------------------------

        :
            // Горелка выключена (BurnerEnable = FALSE по выходам ниже)

            // Барабан крутится — продукт проходит через барабан

            // Вентилятор: при аварии вент — выкл, иначе — работает

            TonUnload(IN := ,
                       PT := DataHMI.UnloadTime);
            
            IF TonUnload.Q THEN
                RESET_TIMER(TonUnload);
                DrumReady := ;
                Step := ;
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 7: ОХЛАЖДЕНИЕ

            // -------------------------------------------------------

        :
            // Горелка выкл, барабан и вентилятор крутятся

            TonCooldown(IN := ,
                         PT := DataHMI.CooldownTime);
            
            IF TonCooldown.Q THEN
                RESET_TIMER(TonCooldown);
                // Запускаем таймер "барабан ещё тёплый"

                WarmAfterStop := ;
                Step := ;
            END_IF;
            
            // -------------------------------------------------------

            // ШАГ 8: АВАРИЯ

            // -------------------------------------------------------

        :
            // Всё оборудование выключено (по выходам ниже)

            // Сброс только оператором вручную

            IF DataHMI.AlarmReset THEN
                DataHMI.AlarmReset := ;
                DataHMI.StepAlarmCode := ;
                WarmAfterStop := ;
                Step := ;
            END_IF;
            
    END_CASE;
END_REGION

    // Вентилятор: работает во всех шагах кроме IDLE, FAULT

    // и EMERGENCY_UNLOAD при аварии вентилятора

    FanStart := (Step = ) OR
    (Step = ) OR
    (Step = ) OR
    (Step = ) OR
    (Step = ) OR

FUNCTION_BLOCK "DrumTempControl"  // SCL  Block #24

VAR_INPUT
    TempOut : Real;  // Температура на выходе барабана, °C
    TempIn : Real;  // Температура на входе барабана, °C
    Enable : Bool;  // Разрешение работы блока
    AlarmReset : Bool;  // Сброс аварии оператором
    TempSetpoint : Real;  // Уставка температуры на выходе, °C
    TempHysteresis : Real;  // Гистерезис, °C (например 15.0)
    TempInLimit : Real;  // Максимальная допустимая T на входе, °C
END_VAR
VAR_OUTPUT
    UpPower : Bool;  // Увеличить мощность горелки
    BurnerEnable : Bool;  // Разрешение работы горелки
    TempInAlarm : Bool;  // Авария: T_in превысила лимит и не снизилась
END_VAR
VAR
    TonTempIn : TON_TIME;
    TempInHighActive : Bool;  // Флаг: T_in сейчас выше лимита
END_VAR
VAR CONSTANT
    TempInHysteresis : Real := 5.0;  // Гистерезис снятия флага превышения T_in, °C
    TempInAlarmDelay : Time := T#1m;  // Таймер ожидания снижения T_in
END_VAR

// ==================== CODE ====================
// Network 1
// Блок регулирования мощности горелки сушильного барабана.
    Входные сигналы:
        TempOut          - температура на выходе барабана (4..20 мА, Real °C)
        TempIn           - температура на входе барабана  (4..20 мА, Real °C)
        Enable           - разрешение работы блока (от блока верхнего уровня)
        AlarmReset       - сброс аварии оператором
    Уставки передаются из DataHMI вызывающего блока:
        TempSetpoint     - уставка температуры на выходе барабана
        TempHysteresis   - зона нечувствительности (гистерезис)
        TempInLimit      - максимально допустимая температура на входе

REGION Регулирование мощности по T_out (гистерезис)
    // Работаем только если блок разрешён и нет аварии

    IF Enable AND NOT TempInAlarm THEN
        
        IF TempOut < (TempSetpoint - TempHysteresis) THEN
            // Температура ниже нижней границы зоны — греем

            UpPower := ;
            
        ELSIF TempOut > (TempSetpoint + TempHysteresis) THEN
            // Температура выше верхней границы зоны — снижаем мощность

            UpPower := ;
            
            // ELSE: зона нечувствительности — состояние UpPower не меняем

        END_IF;
        
    ELSE
        // Блок выключен или авария — снимаем сигнал увеличения мощности

        UpPower := ;
    END_IF;
END_REGION
REGION Контроль температуры на входе барабана
    // Фиксируем факт превышения T_in

    IF TempIn > TempInLimit THEN
        TempInHighActive := ;
    END_IF;
    
    // Снимаем флаг только если T_in вернулась в норму (с гистерезисом)

    IF TempInHighActive AND (TempIn < (TempInLimit - )) THEN
        TempInHighActive := ;
    END_IF;
    
    // Запускаем таймер пока T_in превышена

    TonTempIn(IN := TempInHighActive AND Enable,
               PT := );
    
    // Авария фиксируется если таймер истёк — сброс только оператором

    IF TonTempIn.Q THEN
        TempInAlarm := ;
    END_IF;
    
    // Сброс аварии оператором — только если T_in уже в норме

    IF AlarmReset AND NOT TempInHighActive THEN
        TempInAlarm := ;
        RESET_TIMER(TonTempIn);
    END_IF;
END_REGION
REGION Разрешение горелки
    // Горелка разрешена если: блок активен, нет аварии T_in, T_in не превышена прямо сейчас

    BurnerEnable := Enable AND NOT TempInAlarm AND NOT TempInHighActive;
END_REGION

VAR_INPUT

    RightLimit : Bool;  // концевой выключатель справа (силос 8)
    Operator_Silo : Int;  // выбранный силос (1..8)
    StartCmd : Bool;  // команда "Пуск" (фронт)
    StopCmd : Bool;  // команда "Стоп"
    Reset : Bool;  // сброс аварии

    Drive_Dir : Int;  // 0=стоп, 1=вперёд (к силосу 8), 2=назад (к силосу 1)
    Drive_Freq : Real;  // частота задания ПЧ (Гц)
    AtTarget : Bool;  // флаг достижения цели

    last_known_idx : Int := -1;  // последний известный индекс

    pending_target : Int;  // отложенная цель, если ушли в хоуминг

    W : Int;
    X : Int;
    Y : Int;
    Z : Int;

    STATE_IDLE : Int;
    STATE_MOVING : Int := 1;
    STATE_HOMING : Int := 2;
    STATE_ESTOP : Int := 3;

// Инициализация

    Error := ;
    ErrorCode := ;
    AtTarget := ;
    Drive_Dir := ;
    Drive_Freq := ;
    RESET_TIMER(move_timer);
    homing_tried_both := ;
    RETURN;

// -------------------------------------------------------------------------

// Вычисление кода датчиков (0..15) для доступа к DecodeTable

IF Sensor_W THEN
    W := ;
ELSE
    W := ;

IF Sensor_X THEN
    X := ;
ELSE
    X := ;

    Y := ;
ELSE
    Y := ;

    Z := ;
ELSE
    Z := ;

// -------------------------------------------------------------------------

// Декодирование текущей позиции по сигналам датчиков

// -------------------------------------------------------------------------

current_idx := Config.DecodeTable[sensorCode];
// Обновление последней известной позиции

IF current_idx <>  THEN
    last_known_idx := current_idx;
END_IF;
// -------------------------------------------------------------------------

// Конечный автомат

// -------------------------------------------------------------------------

CASE state OF
        
        // ---------------------------------------------------------------------

        // STATE_IDLE – ожидание команды

        // ---------------------------------------------------------------------

    :
        Error := ;
        ErrorCode := ;
        AtTarget := ;
        Drive_Dir := ;
        Drive_Freq := ;
        RESET_TIMER(move_timer);
        
        IF (Config.SiloTarget[Operator_Silo] >= Config.MinIndex AND Config.SiloTarget[Operator_Silo] <= Config.MaxIndex) THEN
            // Определить целевой индекс по выбранному силосу

            target_idx := Config.SiloTarget[Operator_Silo];
            IF current_idx <>  THEN
                // Позиция известна

                IF target_idx = current_idx THEN
                    AtTarget := ;
                ELSE
                    state := ;
                    // Направление

                    IF target_idx > current_idx THEN
                        direction := ;
                    ELSE
                        direction := ;
                    END_IF;
                    // Получаем точку торможения из конфигурации

                    IF direction =  THEN
                        brake_idx := Config.BrakePoints[target_idx].leftBrake;
                    ELSE
                        brake_idx := Config.BrakePoints[target_idx].rightBrake;
                    END_IF;
                END_IF;
            ELSE
                // Позиция неизвестна – запоминаем цель и ищем

                pending_target := target_idx;
                homing_tried_both := ;
                state := ;
            END_IF;
        END_IF;
        
        // ---------------------------------------------------------------------

        // STATE_MOVING – движение к заданной цели

        // ---------------------------------------------------------------------

    :
        AtTarget := ;
        
        // Если позиция известна, проверяем достижение цели и управляем скоростью

        IF current_idx <>  THEN
            IF current_idx = target_idx THEN
                // Достигли цели

                Drive_Dir := ;
                Drive_Freq := ;
                AtTarget := ;
                state := ;
            ELSE
                // Управление скоростью: если достигли точки торможения – снижаем

                IF current_idx = brake_idx THEN
                    Drive_Freq := ;
                ELSE
                    Drive_Freq := ;
                END_IF;
                Drive_Dir := direction;
            END_IF;
            // Позиция неизвестна (между пластинами) – продолжаем движение с прежней скоростью и направлением

            // Скорость остаётся той, которая была установлена ранее (высокая или низкая – сохраняем)

            // Для простоты оставляем #Drive_Freq как есть (оно уже было установлено в предыдущем цикле)

            // Направление тоже не меняем

        END_IF;
        
        // Контроль времени движения (таймер работает всегда, когда мы в MOVING)

        move_timer(IN := ,
                    PT := );
        IF move_timer.Q THEN
            state := ;
            Error := ;
            ErrorCode := ;   // тайм-аут движения

            Drive_Dir := ;
            Drive_Freq := ;
        END_IF;
        
        
        // ---------------------------------------------------------------------

        // STATE_HOMING – поиск позиции (движение с малой скоростью)

        // ---------------------------------------------------------------------

    :
        AtTarget := ;
        
        // Если позиция появилась

        IF current_idx <>  THEN
            Drive_Dir := ;
            Drive_Freq := ;
            homing_tried_both := ;
            
            IF pending_target <>  THEN
                target_idx := pending_target;
                pending_target := ;
                state := ;
                IF target_idx > current_idx THEN
                    direction := ;
                ELSIF target_idx < current_idx THEN
                    direction := ;
                ELSE
                    AtTarget := ;
                    state := ;
                END_IF;
                IF direction =  THEN
                    brake_idx := Config.BrakePoints[target_idx].leftBrake;
                ELSE
                    brake_idx := Config.BrakePoints[target_idx].rightBrake;
                END_IF;
            ELSE
                state := ;
            END_IF;
        ELSE
                // Продолжаем движение с малой скоростью

            // Проверяем последнюю позцицию и вычисляем куда двигаться

            IF NOT homing_tried_both THEN
                IF last_known_idx <>  THEN
                    IF last_known_idx <= (Config.MinIndex + Config.MaxIndex) /  THEN
                        homing_dir := ;
                    ELSE
                        homing_dir := ;
                    END_IF;
                ELSE
                    homing_dir := ; // просто движемся вправо

                END_IF;
            END_IF;
            // Проверка концевиков

            IF (homing_dir =  AND RightLimit) OR
                (homing_dir =  AND LeftLimit) THEN
                IF NOT homing_tried_both THEN
                    // меняем направление

                    IF homing_dir =  THEN
                        homing_dir := ;
                    ELSIF homing_dir =  THEN
                        homing_dir := ;
                    END_IF;
                    homing_tried_both := ;
                ELSE
                    state := ;
                    Error := ;
                    ErrorCode := ;   // нет опорных позиций

                    Drive_Dir := ;
                    Drive_Freq := ;
                END_IF;
            END_IF;
            // Управление

            Drive_Dir := homing_dir;
            Drive_Freq := ;
            move_timer(IN := ,
                        PT := );
            IF move_timer.Q THEN
                state := ;
                Error := ;
                ErrorCode := ;   // тайм-аут хоуминга

                Drive_Dir := ;
                Drive_Freq := ;
            END_IF;
        END_IF;
        
        // ---------------------------------------------------------------------

        // STATE_ESTOP – аварийная остановка

        // ---------------------------------------------------------------------

    :
        Drive_Dir := ;
        Drive_Freq := ;
        AtTarget := ;
        IF Reset AND NOT LeftLimit AND NOT RightLimit THEN
            state := ;
            Error := ;
            ErrorCode := ;
        END_IF;
END_CASE;
// -------------------------------------------------------------------------

// Принудительная остановка по команде оператора

// -------------------------------------------------------------------------

IF StopCmd AND state <>  THEN
    state := ;
    Drive_Dir := ;
    Drive_Freq := ;
    AtTarget := ;
    RESET_TIMER(move_timer);
END_IF;

FUNCTION_BLOCK "Control_Speed"  // SCL  Block #6

VAR_INPUT

    CycleTime : Time;  // Период вызова блока

    WatchdogTime : Time;  // Максимальное время без импульсов
END_VAR
VAR_OUTPUT
    Speed : Real;  // Текущая скорость (м/с)
    Speed_m_min : Real;  // Скорость в м/мин
    PulsesPerInterval : DInt;  // Количество импульсов за последний интервал
    Valid : Bool;  // TRUE, если данные валидны (прошёл хотя бы один интервал)
    Feedback : Bool;  // Обратная связь с датчика
END_VAR
VAR
    pulseCnt : DInt;
    callCnt : DInt;
    cyclesPerAvg : DInt;
    initDone : Bool;
    watchdog : Bool;  // Сигнал проверки приходящего сигнала
    watchdogTimer : TON_TIME;
    WatchdogAlarm : Bool;

VAR CONSTANT
    AveragingTime : Time := T#1s;  // Интервал усреднения скорости
END_VAR

// ==================== CODE ====================
// Network 1
// Проверка необходимсоти использования блока

IF NOT Enable THEN
    Speed := ;
    Speed_m_min := ;
    Valid := ;
    PulsesPerInterval := ;
    Feedback := ;
    initDone := ;
    WatchdogAlarm := ;
    RETURN;
END_IF;
// Обратная связь по работе счетчика

IF NOT WatchdogAlarm THEN
    Feedback := ;
ELSE
    Feedback := ;
END_IF;
// Инициализация: вычисляем количество вызовов на интервал усреднения

IF NOT initDone THEN
    cyclesPerAvg := DINT_TO_REAL(TIME_TO_DINT()) / DINT_TO_REAL(TIME_TO_DINT(CycleTime));
    initDone := ;
END_IF;
// Счетчик импульсов

IF SensorSignal THEN
    pulseCnt := pulseCnt + ;
    watchdog := ;
ELSE
    watchdog := ;
END_IF;
// Таймер работы счетчика

watchdogTimer(IN := watchdog,
               PT := WatchdogTime,
               Q => WatchdogAlarm
);
// Счётчик вызовов

callCnt := callCnt + ;
// Если наступило время обновления скорости

IF callCnt >= cyclesPerAvg THEN
    // Сохраняем число импульсов за интервал

    PulsesPerInterval := pulseCnt;
    
    // Вычисляем скорость (м/с)

    Speed := DINT_TO_REAL(pulseCnt) * LengthPerPulse;
    Speed_m_min := Speed * ;
    
    Valid := ;
    
    // Сброс счётчиков

    pulseCnt := ;
    callCnt := ;
END_IF;

FUNCTION_BLOCK "V_TRIG"  // SCL  Block #25

VAR_INPUT
    IN : Variant;
END_VAR
VAR_OUTPUT
    Q : Bool;
    R : Bool;
    F : Bool;
END_VAR
VAR
    mInt : Int;
    mReal : Real;
END_VAR
VAR_TEMP
    retVal : Int;
    DataType : UInt;
    tInt : Int;
    tReal : Real;
END_VAR
VAR CONSTANT
    TYPE_INT : UInt := 1;
    TYPE_REAL : UInt := 2;
END_VAR

// ==================== CODE ====================
// Network 1
REGION Проверка типа входных данных
    IF IS_NULL(IN) THEN
        RETURN;
    ELSIF TypeOf(IN) = TypeOf(mInt) THEN
        DataType := ;
        retVal := MOVE_BLK_VARIANT(SRC := IN, COUNT := , SRC_INDEX := , DEST_INDEX := , DEST => tInt);
    ELSIF TypeOf(IN) = TypeOf(mReal) THEN
        DataType := ;
        retVal := MOVE_BLK_VARIANT(SRC := IN, COUNT := , SRC_INDEX := , DEST_INDEX := , DEST => tReal);
    END_IF;
END_REGION
REGION Проверка изменения величины
    IF mInt > tInt THEN
        Q := R := ;
        F := ;
    ELSIF mInt < tInt THEN
        Q := F := ;
        R := ;
    ELSE
        Q := R := F := ;
    END_IF;
END_REGION
REGION Запоминание нового значения
    CASE DataType OF
        :
            mInt := tInt;
        :
            mReal := tReal;
    END_CASE;
END_REGION
